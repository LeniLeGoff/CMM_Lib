48 std::vector<std::vector<Pixel>> read_mnist_image_file(const std::string& path) {
49  auto buffer = read_mnist_file(path, 0x803);
50 
51  if (buffer) {
52  auto count = read_header(buffer, 1);
53  auto rows = read_header(buffer, 2);
54  auto columns = read_header(buffer, 3);
55 
56  //Skip the header
57  //Cast to unsigned char is necessary cause signedness of char is
58  //platform-specific
59  auto image_buffer = reinterpret_cast<unsigned char*>(buffer.get() + 16);
60 
61  std::vector<std::vector<Pixel>> images;
62  images.reserve(count);
63 
64  for (size_t i = 0; i < count; ++i) {
65  images.emplace_back(rows * columns);
66 
67  for (size_t j = 0; j < rows * columns; ++j) {
68  auto pixel = *image_buffer++;
69  images[i][j] = static_cast<Pixel>(pixel);
70  }
71  }
72 
73  return images;
74  }

76  return {};
77 }
